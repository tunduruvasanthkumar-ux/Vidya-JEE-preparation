        // Update progress circle based on time of day
        function updateProgress() {
            const now = new Date();
            const hours = now.getHours();
            const minutes = now.getMinutes();
            
            // Calculate progress based on time (assuming 9 AM to 9 PM study day)
            let progress = 0;
            if (hours >= 9 && hours < 21) {
                const totalMinutes = 12 * 60; // 12 hours
                const currentMinutes = (hours - 9) * 60 + minutes;
                progress = Math.min(100, Math.round((currentMinutes / totalMinutes) * 100));
            } else if (hours >= 21) {
                progress = 100;
            }
